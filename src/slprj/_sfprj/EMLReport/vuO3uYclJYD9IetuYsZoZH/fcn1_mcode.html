1function y = fcn(qpsk_Q_tx, qpsk_I_tx)
 2%-------------Generation du signal QPSK
 3%%%%%%%%%%%%%%%%%%% Variables à utiliser %%%%%%%%%%%%%%%%%%%%%
 4
 5nsb = 30		%Nombre de bits
 6
 7fb = 1e5		%Fréquence des bits
 8Tb = 1/fb		%Période des bits
 9Eb = 1			%Energie par bit
10
11T = 2*Tb		%Période des symboles
12Nsymb = nsb/2   %Nombre de symboles
13
14fc = 1e6		%Fréquence de la porteuse (carrier)
15
16Ts = 1/(10*fc)	%Période d'échantillonnage : définit le pas de simulation
17fs = 1/Ts		%Fréquence d'échantillonnage
18
19npb = round(Tb/Ts)		%Nombre de points par bit
20
21t = [Ts:Ts:nsb*Tb];     %Vecteur représentant l'axe du temps
22qpsk = 2*sqrt(Eb/T).* cos(2*pi*fc.*t - qpsk_Q_tx.*(pi/2 - qpsk_I_tx*pi/4));
23y = qpsk;
